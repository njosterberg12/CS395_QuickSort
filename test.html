Algorithm Pseudocode:
HoarePartition(A, low, high)
 pivot := A[low]
 i := low
 j := high
 repeat forever
  while A[i]<=pivot && i<high
   i++
  while A[j]>=pivot && j > low
   j--;
  if i < j
   swap A[i] with A[j]
  else
   swap A[low] with A[j]
   return j
